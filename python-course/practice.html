练习1：计算器

创建一个简单的计算器，能够执行基本的算术运算（加、减、乘、除）。 

def calculator():
    """简单计算器"""
    print("简单计算器")
    print("操作：1. 加法 2. 减法 3. 乘法 4. 除法")
    
    # 获取用户输入
    choice = input("请选择操作（1-4）: ")
    num1 = float(input("请输入第一个数: "))
    num2 = float(input("请输入第二个数: "))
    
    # 执行相应的操作
    if choice == "1":
        result = num1 + num2
        print(f"{num1} + {num2} = {result}")
    elif choice == "2":
        result = num1 - num2
        print(f"{num1} - {num2} = {result}")
    elif choice == "3":
        result = num1 * num2
        print(f"{num1} * {num2} = {result}")
    elif choice == "4":
        if num2 != 0:
            result = num1 / num2
            print(f"{num1} / {num2} = {result}")
        else:
            print("错误：除数不能为零")
    else:
        print("错误：无效的选择")

# 调用计算器
calculator()

练习2：猜数字游戏

创建一个猜数字游戏，程序随机生成一个1-100之间的数字，用户需要猜测这个数字，程序会提示用户猜大了还是猜小了，直到猜对为止。 

import random

def guess_number():
    """猜数字游戏"""
    print("猜数字游戏")
    print("我想了一个1-100之间的数字，你能猜对吗？")
    
    # 生成随机数字
    secret_number = random.randint(1, 100)
    guess = 0
    attempts = 0
    
    while guess != secret_number:
        guess = int(input("请输入你的猜测: "))
        attempts += 1
        
        if guess < secret_number:
            print("猜小了！")
        elif guess > secret_number:
            print("猜大了！")
        else:
            print(f"恭喜你猜对了！用了{attempts}次尝试。")

# 调用猜数字游戏
guess_number()

练习3：学生成绩管理

创建一个学生成绩管理系统，能够添加学生信息、查看学生信息、计算平均分等。 

def student_grade_management():
    """学生成绩管理系统"""
    students = []
    
    while True:
        print("\n学生成绩管理系统")
        print("1. 添加学生")
        print("2. 查看所有学生")
        print("3. 计算平均分")
        print("4. 退出")
        
        choice = input("请选择操作: ")
        
        if choice == "1":
            # 添加学生
            name = input("请输入学生姓名: ")
            math = float(input("请输入数学成绩: "))
            english = float(input("请输入英语成绩: "))
            chinese = float(input("请输入语文成绩: "))
            
            student = {
                "name": name,
                "math": math,
                "english": english,
                "chinese": chinese,
                "average": (math + english + chinese) / 3
            }
            
            students.append(student)
            print(f"学生 {name} 添加成功！")
            
        elif choice == "2":
            # 查看所有学生
            if not students:
                print("暂无学生信息")
            else:
                print("\n学生信息:")
                for i, student in enumerate(students, 1):
                    print(f"{i}. 姓名: {student['name']}, 数学: {student['math']}, 英语: {student['english']}, 语文: {student['chinese']}, 平均分: {student['average']:.2f}")
            
        elif choice == "3":
            # 计算平均分
            if not students:
                print("暂无学生信息")
            else:
                total_math = sum(student['math'] for student in students)
                total_english = sum(student['english'] for student in students)
                total_chinese = sum(student['chinese'] for student in students)
                total_average = sum(student['average'] for student in students)
                
                print("\n班级平均分:")
                print(f"数学: {total_math / len(students):.2f}")
                print(f"英语: {total_english / len(students):.2f}")
                print(f"语文: {total_chinese / len(students):.2f}")
                print(f"总平均分: {total_average / len(students):.2f}")
            
        elif choice == "4":
            # 退出
            print("退出系统")
            break
        else:
            print("错误：无效的选择")

# 调用学生成绩管理系统
student_grade_management()

练习4：单词计数器

创建一个单词计数器，能够统计文本中每个单词出现的次数。 

def word_counter():
    """单词计数器"""
    text = input("请输入一段文本: ")
    
    # 处理文本
    text = text.lower()
    # 移除标点符号
    punctuation = "!"#$%&'()*+,-./:;<=>?@[\]^_`{|}~"
    for char in punctuation:
        text = text.replace(char, "")
    
    # 分割单词
    words = text.split()
    
    # 统计单词出现次数
    word_count = {}
    for word in words:
        if word in word_count:
            word_count[word] += 1
        else:
            word_count[word] = 1
    
    # 显示结果
    print("\n单词统计结果:")
    print(f"总单词数: {len(words)}")
    print(f"不同单词数: {len(word_count)}")
    print("\n单词出现次数:")
    for word, count in sorted(word_count.items()):
        print(f"{word}: {count}")

# 调用单词计数器
word_counter()
